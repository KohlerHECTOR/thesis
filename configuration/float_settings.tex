\usepackage{float}

% Better float placement parameters
\renewcommand{\topfraction}{0.85}        % Max fraction of page for floats at top
\renewcommand{\bottomfraction}{0.75}     % Max fraction of page for floats at bottom
\renewcommand{\textfraction}{0.15}       % Min fraction of page for text
\renewcommand{\floatpagefraction}{0.7}   % Min fraction of page for first float on float page
\renewcommand{\dbltopfraction}{0.85}     % Max fraction of page for float at top of 2-col page
\renewcommand{\dblfloatpagefraction}{0.7}% Min fraction of page for first float on 2-col float page
\setcounter{topnumber}{3}               % Max number of floats at top of page
\setcounter{bottomnumber}{3}            % Max number of floats at bottom of page
\setcounter{totalnumber}{5}             % Max total number of floats on a page
\setcounter{dbltopnumber}{2}            % Max number of 2-col floats at top of 2-col page

% Prevent widows and orphans
\widowpenalty=10000
\clubpenalty=10000

% Allow more flexibility in float placement
\setlength{\textfloatsep}{10pt plus 4pt minus 4pt}     % Space between floats and text
\setlength{\floatsep}{10pt plus 4pt minus 4pt}         % Space between consecutive floats
\setlength{\intextsep}{10pt plus 4pt minus 4pt}        % Space between in-text floats and text
